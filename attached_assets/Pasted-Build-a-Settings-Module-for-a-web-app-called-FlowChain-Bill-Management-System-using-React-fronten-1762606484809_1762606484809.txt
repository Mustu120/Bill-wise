üì¶ PostgreSQL Schema
CREATE TABLE partners (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255),
  type VARCHAR(20) CHECK (type IN ('customer','vendor','both')) DEFAULT 'both'
);

CREATE TABLE projects (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255)
);

CREATE TABLE products (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255),
  for_sales BOOLEAN DEFAULT FALSE,
  for_purchase BOOLEAN DEFAULT FALSE,
  for_expenses BOOLEAN DEFAULT FALSE,
  sales_price NUMERIC(12,2) DEFAULT 0,
  cost NUMERIC(12,2) DEFAULT 0
);

CREATE TABLE taxes (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100),
  rate NUMERIC(6,3)
);

CREATE TABLE sales_orders (
  id SERIAL PRIMARY KEY,
  code VARCHAR(30) UNIQUE,
  customer_id INT REFERENCES partners(id),
  project_id INT REFERENCES projects(id),
  status VARCHAR(20) DEFAULT 'Draft',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE sales_order_lines (
  id SERIAL PRIMARY KEY,
  order_id INT REFERENCES sales_orders(id),
  product_id INT REFERENCES products(id),
  quantity NUMERIC(12,3),
  unit VARCHAR(20),
  unit_price NUMERIC(12,2),
  tax_ids INT[],
  amount NUMERIC(12,2)
);

CREATE TABLE purchase_orders (
  id SERIAL PRIMARY KEY,
  code VARCHAR(30) UNIQUE,
  vendor_id INT REFERENCES partners(id),
  project_id INT REFERENCES projects(id),
  status VARCHAR(20) DEFAULT 'Draft'
);

CREATE TABLE invoices (
  id SERIAL PRIMARY KEY,
  number VARCHAR(40),
  type VARCHAR(20) CHECK (type IN ('customer','vendor')),
  partner_id INT REFERENCES partners(id),
  project_id INT REFERENCES projects(id),
  status VARCHAR(20) DEFAULT 'Draft'
);

CREATE TABLE invoice_lines (
  id SERIAL PRIMARY KEY,
  invoice_id INT REFERENCES invoices(id),
  product_id INT REFERENCES products(id),
  quantity NUMERIC(12,3),
  unit_price NUMERIC(12,2),
  tax_ids INT[],
  amount NUMERIC(12,2)
);

CREATE TABLE expenses (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255),
  project_id INT REFERENCES projects(id),
  period_start DATE,
  period_end DATE,
  image_url TEXT,
  description TEXT,
  ocr_data JSONB,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
